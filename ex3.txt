Function to print array with indentation (diamond effect)
void printArray(int arr[], int n, int depth)
{
    Indentation spaces
    for (int i = 0; i < depth; i++)
        cout << "   ";

    cout << "[ ";
    for (int i = 0; i < n; i++)
        cout << arr[i] << " ";
    cout << "]" << endl;
}

Merge function
void merge(int B[], int p, int C[], int q, int A[])
{
    int i = 0, j = 0, k = 0;

    while (i < p && j < q)
    {
        if (B[i] <= C[j])
            A[k++] = B[i++];
        else
            A[k++] = C[j++];
    }

    while (i < p)
        A[k++] = B[i++];

    while (j < q)
        A[k++] = C[j++];
}

Merge Sort with Diamond Printing
void mergesort(int A[], int n, int depth = 0)
{
    if (n > 1)
    {
        Print splitting step (going down)
        for (int i = 0; i < depth; i++)
            cout << "   ";
        cout << "Splitting: ";
        printArray(A, n, 0);

        int mid = n / 2;

        int B[mid];
        int C[n - mid];

         Copy into B
        for (int i = 0; i < mid; i++)
            B[i] = A[i];

         Copy into C
        for (int i = mid; i < n; i++)
            C[i - mid] = A[i];

         Recursive calls
         mergesort(B, mid, depth + 1);
         mergesort(C, n - mid, depth + 1);

         Merge step
         merge(B, mid, C, n - mid, A);

         Print merging step (coming up)
        for (int i = 0; i < depth; i++)
            cout << "   ";
        cout << "Merging : ";
        printArray(A, n, 0);
    }
}

Partition function (numbers only)
int partition(int A[], int l, int r, int n, int depth)
{
    int p = A[l];
    int i = l;
    int j = r + 1;

    while(true)
    {
        do { i++; } while(A[i] < p);
        do { j--; } while(A[j] > p);

        if(i >= j)
            break;

        swap(A[i], A[j]);
        printArray(A, n, depth);
    }

    swap(A[l], A[j]);
    printArray(A, n, depth);

    return j;
}

Quick Sort with skeletal depth
void quickSort(int A[], int l, int r, int n, int depth)
{
    if(l < r)
    {
        int s = partition(A, l, r, n, depth);
        quickSort(A, l, s - 1, n, depth + 1);
        quickSort(A, s + 1, r, n, depth + 1);
    }
}

int main()
{
    int A[] = {7, 4, 3, 10};
    int n = sizeof(A) / sizeof(A[0]);

    printArray(A, n, 0);
    quickSort(A, 0, n - 1, n, 0);

    return 0;
}
