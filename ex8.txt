#include <iostream>
#include <vector>
#include <cmath>

using namespace std;

int x[20];
int solutionCount = 0;

void printSolutionMatrix(int n) {
    solutionCount++;
    cout << "Solution #" << solutionCount << endl;
    cout << "Numbers: [ ";
    for (int j = 1; j <= n; j++) {
        cout << x[j] << " ";
    }
    cout << "]" << endl;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            if (x[i] == j)
                cout << "Q ";
            else
                cout << " * ";
        }
        cout << endl;
    }
}

bool Place(int k, int i) {
    for (int j = 1; j <= k - 1; j++) {
        if ((x[j] == i) || (abs(x[j] - i) == abs(j - k))) {
            return false;
        }
    }
    return true;
}

void NQueens(int k, int n) {
    for (int i = 1; i <= n; i++) {
        if (Place(k, i)) {
            x[k] = i;
            if (k == n) {
                printSolutionMatrix(n);
            } else {
                NQueens(k + 1, n);
            }
        }
    }
}

int main() {
    int n;
    cout << "Enter the number of queens (n): ";
    cin >> n;

    if (n > 0 && n < 20) {
        NQueens(1, n);
        if (solutionCount == 0) cout << "No solutions found." << endl;
    } else {
        cout << "enter a valid number (1-19)." << endl;
    }

    return 0;
}
