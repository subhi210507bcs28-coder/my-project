#include<iostream>
#include<cmath>
using namespace std;
struct Point {
   double x,y;
};
double closestpair(Point P[],int n)
{
   double d=9999.99;
   for(int i=0;i<n-1;i++)
   {
      for(int j=i+1;j<n;j++)
      {
         double dist=sqrt((P[i].x-P[j].x)*(P[i].x-P[j].x)+(P[i].y-P[j].y)*(P[i].y-P[j].y));
         cout <<"Distance between P"<<i+1<<"("<<P[i].x<<","<<P[i].y<<")and P"<<j+1<<"("<<P[j].x<<","<<P[j].y<<")="<<dist<<endl;
         if (dist < d)
         {
            d=dist;
            cout<<"------------------------------\n";
            cout<<"New minimum distance="<<d<<endl;
            cout<<"------------------------------\n";
         }
      }
   }
   return d;
}
int main() {
   int n;
   cout<<"Enter number of points:";
   cin>>n;
   Point P[n];
   cout<<"Enter Points(x,y):"<<endl;
   for(int i=0;i<n;i++) {
      cin>>P[i].x>>P[i].y;
   }
   double result=closestpair(P,n);
   cout<<"------------------------------\n";
   cout<<"Final Minimum distance:"<<result<<endl;
   cout<<"------------------------------\n";
   return 0;
}

#include<iostream>
#include<string>
using namespace std;

void bruteforce_visual(string T, string P) {
    int n = T.length();
    int m = P.length();
    int total_comparisons = 0;
    int total_alignments = 0;

    for (int i = 0; i <= n - m; i++) {
        total_alignments++;
        cout << "Shift " << i << ":" << endl;
        cout << "Text:    " << T << endl;
        cout << "Pattern: ";
        for (int k = 0; k < i; k++) cout << " ";
        cout << P << endl;

        cout << "Status:  ";
        for (int k = 0; k < i; k++) cout << " ";

        int j = 0;
        while (j < m) {
            total_comparisons++;
            if (T[i + j] == P[j]) {
                cout << "|";
                j++;
            } else {
                cout << "*";
                break;
            }
        }
        cout << "\n" << endl;

        if (j == m) {
            cout<<"------------------------------\n";
            cout << "Pattern found at index: " << i << endl;
            cout << "Total alignments: " << total_alignments << endl;
            cout << "Total comparisons: " << total_comparisons << endl;
            cout<<"------------------------------\n";
            return;
        }
    }
    cout << "Pattern not found" << endl;
    cout << "Total alignments made: " << total_alignments << endl;
    cout << "Total comparisons made: " << total_comparisons << endl;
}

int main() {
    string T, P;
    cout << "Enter Text: "; getline(cin, T);
    cout << "Enter Pattern: "; getline(cin, P);
    cout << endl;
    bruteforce_visual(T, P);
    return 0;
}
